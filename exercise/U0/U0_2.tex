\begin{algorithm}
    \NoCaptionOfAlgo
    \caption{MSS\_Naive (int[] a, int n)}
    \Begin{
    \SetAlgoVlined
    $maxscore := 0 ; \qquad \ell := 1; \qquad r := 0$\tcc*{ always is maxscore = $\sigma(\ell,r)$}
    \For{$i := 1; i \leq n; i\textbf{\rm\scriptsize++}$}{
        \For{$j := i;j \leq n; j\textbf{\rm\scriptsize++}$}{
            $s := 0$ \tcc*{ compute s = $\sigma(\ell,r)$}\par
        \For{$k := i; k \leq j; k\textbf{\rm\scriptsize++}$}{$s := s + a[k];$}
        \If{$s > maxscore$}{$maxscore := s; \qquad  \ell:= i; \qquad r := j;$}
            }
        }
    }
\end{algorithm}
