\textbf{1. Aufgabe (10 Punkte): Double Digest Problem}\\ \\

 \begin{algorithm}
    \NoCaptionOfAlgo
    \caption{DoubleDigest())}
    \Begin{
    \SetAlgoVlined
       generate random initial population;\\
       repeat;\\
       Natural Selection;\\
       select parents from population;\\
       order preserving weighted crossover on parents;\\
       mutate individuals;\\
       reserve Solution;
       form next generation;\\
       E-class={s};\\
       \While{(unmarked string s1 in E-class)}{
    E-class=E-Class $\cup$Permutation (s1)
       }
       \While{(marked string s2 in E-class)}{
       E-class=E-Class $\cup$Cassette_Transformations(s2)}
       }
\end{algorithm}\\
